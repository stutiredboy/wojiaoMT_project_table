help				显示帮助信息
------------------------------------------------------------------------------
use 				使用另一个数据库，使用数据库名作为参数
	参数：db_name
	db_name			- 使用show databases显示的数据库名
------------------------------------------------------------------------------
show				显示系统当前可用的数据库或者当前可用的表
	参数：databases | tables
	databases 		- 显示当前可用的数据库
	tables			- 显示当前可用的表
------------------------------------------------------------------------------
adddbx 				添加一个数据库，使用数据库路径作为参数
	参数：db_path
	db_path			- 数据库路径
------------------------------------------------------------------------------
removedbx			删除一个数据库，使用数据库名作为参数
	参数：db_name
	db_name			- 使用show databases显示的数据库名
------------------------------------------------------------------------------
extractdbx			从当前数据库中根据条件提取一部分数据生成一个新库
	参数：xdb_xml_path new_db_path condition+
	xdb_xml_path 	- xdb.xml的路径
	new_db_path		- 新库所在路径
	condition		- 条件限制，可以有多个，一个条件形如key_name:key_id
------------------------------------------------------------------------------
select 				
update
delete				xql查询语句，具体语法参考xql指南
------------------------------------------------------------------------------
exit 				退出本系统
